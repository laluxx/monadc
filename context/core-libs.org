  Pattern matches on list cons =[x|xs]=.

[OBS id:obs.coll.map-filter src:core/prelude/Coll.mon:51-53,100-106 conf:high]
  =(map f list)= — applies f to each element. =(filter p list)= —
  keeps elements satisfying predicate.

[OBS id:obs.coll.foldl-foldr src:core/prelude/Coll.mon:69-76 conf:high]
  =(foldl f acc list)= — left-associative fold. =(foldr f acc list)= —
  right-associative fold. These are the core reduction primitives.

[OBS id:obs.coll.takeWhile-dropWhile src:core/prelude/Coll.mon:37-47 conf:high]
  =(takeWhile p list)= — take prefix while predicate holds.
  =(dropWhile p list)= — drop prefix while predicate holds.

[OBS id:obs.coll.zip-zipwith src:core/prelude/Coll.mon:79-88 conf:high]
  =(zip a b)= — pair-wise combining into list of pairs.
  =(zipwith f a b)= — pair-wise combining via function f.

[OBS id:obs.coll.append-filter src:core/prelude/Coll.mon:91-93,100-106 conf:high]
