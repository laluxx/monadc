[OBS id:obs.coll.define-syntax src:Coll.mon:23-105 conf:high]
  Uses alternative define syntax =(define fn :: Type ...)= with pattern
  matching clauses: =_ [] -> []=, =n [x|xs] | n > 0 -> ...=. This is
  the pure-functional subset of Monad (no =(define (name args))=).

[OBS id:obs.coll.take-drop src:Coll.mon:23-34 conf:high]
  =(take n list)= and =(drop n list)= — basic sequence prefix/suffix.
  Pattern matches on list cons =[x|xs]=.

[OBS id:obs.coll.map-filter src:Coll.mon:51-53,100-106 conf:high]
  =(map f list)= — applies f to each element. =(filter p list)= —
  keeps elements satisfying predicate.

[OBS id:obs.coll.foldl-foldr src:Coll.mon:69-76 conf:high]
  =(foldl f acc list)= — left-associative fold. =(foldr f acc list)= —
  right-associative fold. These are the core reduction primitives.

[OBS id:obs.coll.takeWhile-dropWhile src:Coll.mon:37-47 conf:high]
  =(takeWhile p list)= — take prefix while predicate holds.
  =(dropWhile p list)= — drop prefix while predicate holds.

[OBS id:obs.coll.zip-zipwith src:Coll.mon:79-88 conf:high]
  =(zip a b)= — pair-wise combining into list of pairs.
  =(zipwith f a b)= — pair-wise combining via function f.

[OBS id:obs.coll.append-filter src:Coll.mon:91-93,100-106 conf:high]
  =(append list val)= — append element to list (copies structure).
  =(filter p list)= — select matching elements.

[OBS id:obs.coll.both src:Coll.mon:108-109 conf:high]
  =(both f g)= — predicate conjunction, returns =(lambda (x) (and (f x) (g x)))=.

[OBS id:obs.coll.any-all src:Coll.mon:56-66 conf:high]
